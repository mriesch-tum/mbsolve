144 class qm_lindblad_relaxation : public qm_superop

148  std::vector<real> m_dephasing;
149 
150 public:
160  explicit qm_lindblad_relaxation(
161  const std::vector<std::vector<real> >& rates)
162  : qm_superop(rates.size()),
163  m_scattering(rates.size(), std::vector<real>(rates.size(), 0)),
164  m_dephasing((rates.size() * (rates.size() - 1)) / 2, 0)
165  {
166  /* TODO exception or assert: rates not quadratic? */
167  /* TODO exception or assert: all physical constraints fulfilled? */
168  /* TODO in particular pure dephasing issue?! */
169  /* TODO last main diagonal element must be zero? */
170 
175  /* determine relaxation rates */
176  for (int m = 0; m < m_num_levels; m++) {
177  real relaxation_rate = 0;
178 
179  for (int j = 0; j < m_num_levels; j++) {
180  if (j != m) {
181  relaxation_rate += rates[j][m];
182  m_scattering[m][j] = rates[m][j];
183  }
184  }
185  m_scattering[m][m] = -relaxation_rate;
186  }
187 
188  /* determine dephasing rate for transition mn */
189  unsigned int idx_dephasing = 0;
190  for (int n = 0; n < m_num_levels; n++) {
191  for (int m = 0; m < n; m++) {
192  real dephasing = 0;
193 
194  /* dephasing due to relaxation */
195  dephasing -= 0.5 *
196  (std::abs(m_scattering[m][m]) +
197  std::abs(m_scattering[n][n]));
198  /* pure dephasing */
199  dephasing -= 0.5 * (rates[m][m] + rates[n][n]);
200 
201  m_dephasing[idx_dephasing] = dephasing;
202  idx_dephasing++;
203  }
204  }
205  }
