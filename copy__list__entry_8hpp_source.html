116 class copy_list_entry
117 {
118 private:
119  std::shared_ptr<result> m_result;
120  std::shared_ptr<const record> m_record;
121 
122  /* m_dev members */
123 
124  copy_list_entry_dev m_dev;
125 
126 public:
127  copy_list_entry(
128  std::shared_ptr<const record> rec,
129  std::shared_ptr<const scenario> scen,
130  uint64_t offset_scratch)
131  : m_record(rec)
132  {
133  m_dev.m_timestep = scen->get_timestep_size();
134 
135  if (rec->get_interval() <= scen->get_timestep_size()) {
136  m_dev.m_rows = scen->get_num_timesteps();
137  m_dev.m_interval = scen->get_timestep_size();
138  m_dev.m_interval_ratio = 1.0;
139  } else {
140  m_dev.m_rows =
141  floor(scen->get_endtime() / rec->get_interval()) + 1;
142  m_dev.m_interval = rec->get_interval();
143  m_dev.m_interval_ratio =
144  scen->get_timestep_size() / rec->get_interval();
145  }
146 
147  if (rec->get_position() < 0.0) {
148  /* copy complete grid */
149  m_dev.m_position_idx = 0;
150  m_dev.m_cols = scen->get_num_gridpoints();
151  } else {
152  m_dev.m_position_idx =
153  std::round(rec->get_position() / scen->get_gridpoint_size());
154  m_dev.m_cols = 1;
155  }
156 
157  /* create result */
158  m_result = std::make_shared<result>(
159  rec->get_name(), m_dev.m_cols, m_dev.m_rows);
160 
161  m_dev.m_type = rec->get_type();
162  m_dev.m_timestep = scen->get_timestep_size();
163  m_dev.m_is_complex = rec->is_complex();
164 
165  m_dev.m_offset_scratch_real = offset_scratch;
166  m_dev.m_offset_scratch_imag =
167  offset_scratch + m_dev.m_cols * m_dev.m_rows;
168 
169  m_dev.m_col_idx = rec->get_col();
170  m_dev.m_row_idx = rec->get_row();
171  }
